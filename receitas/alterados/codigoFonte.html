        const altura = 1.71;
        const peso = 72;
        
        /*
          Cálculo de IMC
          > Indíce de massa corporal
        */
        
        function calcularImc(altura, peso) {
          // faz o calculo
          const imc = peso / (altura * altura);
          
          if(imc >= 20 && imc < 23) {
            return "Seu imc é ideal";
          }
          else if(imc >= 23) {
            return "Seu imc está acima do ideal";
          }
          else if(imc < 20) {
            return "Seu imc está abaixo do ideal";
          }
        };
        
        calcularImc(altura, peso);
